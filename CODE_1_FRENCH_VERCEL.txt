<script>
    // Récupérer les paramètres Omada depuis l'URL (si présents)
    const urlParams = new URLSearchParams(window.location.search);
    const macAddress = urlParams.get('mac');
    const clientIp = urlParams.get('ip');
    const sessionId = urlParams.get('id');

    const form = document.getElementById('tapasWifiForm');
    const submitBtn = document.getElementById('submitBtn');
    const successMsg = document.getElementById('successMessage');
    const errorMsg = document.getElementById('errorMessage');
    const loadingMsg = document.getElementById('loadingMsg');

    // Remplacez YOUR_VERCEL_URL par votre URL Vercel réelle
    const VERCEL_URL = 'https://YOUR_VERCEL_URL/api/wifi-fr';

    // Gérer la soumission du formulaire
    form.addEventListener('submit', async function(e) {
        e.preventDefault();

        // Récupérer les données du formulaire
        const email = document.getElementById('tapasWifiEmail').value;
        const name = document.getElementById('tapasWifiName').value;

        // Masquer les messages
        successMsg.style.display = 'none';
        errorMsg.style.display = 'none';
        loadingMsg.style.display = 'block';
        submitBtn.disabled = true;

        try {
            // Préparer les données
            const data = {
                email: email,
                name: name,
                mac_address: macAddress || '',
                client_ip: clientIp || '',
                session_id: sessionId || ''
            };

            // Envoyer à Vercel
            const response = await fetch(VERCEL_URL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(data)
            });

            loadingMsg.style.display = 'none';

            if (response.ok) {
                successMsg.style.display = 'block';
                form.reset();
                document.getElementById('tapasWifiEmail').focus();
            } else {
                errorMsg.textContent = '✗ Une erreur s\'est produite. Veuillez réessayer.';
                errorMsg.style.display = 'block';
            }
        } catch (error) {
            loadingMsg.style.display = 'none';
            errorMsg.textContent = '✗ Erreur de connexion. Vérifiez votre connexion Internet.';
            errorMsg.style.display = 'block';
            console.error('Erreur:', error);
        } finally {
            submitBtn.disabled = false;
        }
    });

    // Focus sur l'email
    document.getElementById('tapasWifiEmail').focus();
</script>
